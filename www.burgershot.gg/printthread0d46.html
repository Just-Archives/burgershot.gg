Code:
forward Player_LoadData(playerid);
public Player_LoadData(playerid) {

   if(cache_affected_rows()) {

       cache_get_value_name_int(0, "Cash", E_PLAYER_INFO[playerid][E_PLAYER_CASH]);
   }

   else {
   //Nese rows nuk jane prekur nga funksioni i meparshem
       SendClientMessage(playerid, -1, "...");
   }
   return 1;
}
