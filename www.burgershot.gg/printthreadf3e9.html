HashTag::suicide(playerid){

public OnPlayerText(playerid,text[]){
    GetHashTag(playerid,text,0xCC0000FF,0x00AA00FF,2048); //back_color the same as first chat color
    SendClientMessageToAll(0x00AA00FF,text);
    return 0;
}
